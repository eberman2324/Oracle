1. HEDWPRD -->  Create new DBLINK_AUD_CONFIG user




CREATE USER DBLINK_AUD_CONFIG  IDENTIFIED BY "Aug02#2023"
    DEFAULT TABLESPACE USERS
    TEMPORARY TABLESPACE TEMP
    QUOTA UNLIMITED ON USERS
    PROFILE TRUSTED_ID_NO_EXPIRE
    ACCOUNT UNLOCK
/
GRANT CREATE SESSION TO DBLINK_AUD_CONFIG
/




HEDWPRD ---> Grant select on table(s)


GRANT SELECT ON prod_Dw.cstmcached_cur_claims_ext TO DBLINK_AUD_CONFIG
/

2. Login to HEDWCFG database with AUD_CUSTOM id and create DBlink

alter session set current_schema=AUD_CUSTOM;


CREATE DATABASE LINK DBLINK_HEDWPRD
    CONNECT TO DBLINK_AUD_CONFIG
    IDENTIFIED BY Aug02#2023
    USING 'HEDWPRD'
/



select count(*) from prod_Dw.cstmcached_cur_claims_ext@DBLINK_HEDWPRD where transaction_time between to_date('07012023','mmddyyyy') and to_date('07312023','mmddyyyy');
