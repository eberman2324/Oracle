ADMINISTER KEY MANAGEMENT CREATE KEYSTORE '/orahome/u01/app/oracle/encryption_wallet/' IDENTIFIED BY "F1shF00d";
ADMINISTER KEY MANAGEMENT CREATE AUTO_LOGIN KEYSTORE FROM KEYSTORE '/orahome/u01/app/oracle/encryption_wallet/' IDENTIFIED BY "F1shF00d";

-- Open wallet 12c
ADMINISTER KEY MANAGEMENT SET KEYSTORE OPEN IDENTIFIED BY "F1shF00d";

-- Create master key 12c
ADMINISTER KEY MANAGEMENT SET KEY IDENTIFIED BY "F1shF00d" WITH BACKUP;
ADMINISTER KEY MANAGEMENT SET KEY USING TAG 'workability' IDENTIFIED BY "F1shF00d" WITH BACKUP USING 'workability_backup';
