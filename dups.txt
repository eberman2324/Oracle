SELECT /*+ index_ffs (s) PARALLEL (4) */ QXJ__PARENT,ATTACHMENT_ID FROM PROD.ATTACHMENT_SET_X_ATTACHMENTS
            WHERE ROWID IN (SELECT MIN(ROWID)
                FROM PROD.ATTACHMENT_SET_X_ATTACHMENTS
                    GROUP BY QXJ__PARENT,ATTACHMENT_ID
                        HAVING COUNT(*) >1)

or

select /*+ index_ffs (s) PARALLEL (4) */  qxj__parent, attachment_id, rowid 
from PROD.ATTACHMENT_SET_X_ATTACHMENTS a 
where rowid > 
(select min(rowid) 
 from PROD.ATTACHMENT_SET_X_ATTACHMENTS b
where b.qxj__parent=a.qxj__parent
and b.attachment_id = a.attachment_id) 
order by 1,2
