    DEFAULT TABLESPACE USERS
    TEMPORARY TABLESPACE TEMP
    PROFILE AETNA_SERVICE_ACCOUNT_PROFILE
    ACCOUNT UNLOCK
/

/


CREATE ROLE REPORT_SELECT NOT IDENTIFIED
/
GRANT REPORT_SELECT TO DBLINK_DMDV01_REPORT
/
ALTER USER DBLINK_DMDV01_REPORT DEFAULT ROLE REPORT_SELECT
/
GRANT SELECT ON DRWKAB.BAC161_FEED_RECORD_HISTORY TO REPORT_SELECT
/
